80template <typename PT, std::size_t PD, class M>
81double SilhouetteMethod<PT, PD, M>::computeSilhouetteScore(int k)
82{
83 // Initialization
84 std::vector<Point<PT, PD>> points = (this->m_kMeans).getPoints();
85 MostDistanceClass mdc(points, k);
86 std::vector<CentroidPoint<PT, PD>> &pointerCentroids = (this->m_kMeans).getCentroids();
87
88 // Execute clustering with k clusters
89 mdc.findCentroid(pointerCentroids);
90 (this->m_kMeans).setNumClusters(static_cast<std::size_t>(k));
91 (this->m_kMeans).fit();
92
93 points = (this->m_kMeans).getPoints();
94
95 double totalScore = 0.0;
96 int numPoints = points.size();
97
104#pragma omp parallel for reduction(+ : totalScore)
105 for (int i = 0; i < numPoints; ++i)
106 {
107 double a = 0.0, c = 0.0;
108 int countA = 0, countC = 0;
109
110 // Compute a(i): the average distance from point i to all other points in the same cluster
111 for (int j = 0; j < numPoints; ++j)
112 {
113 if (i != j && *(points[j].centroid) == *(points[i].centroid))
114 {
115 a += EuclideanMetric<PT, PD>::distanceTo(points[j], points[i]);
116 countA++;
117 }
118 }
119 if (countA > 0)
120 a /= countA; // Normalize a(i)
121
122 // Compute c(i): the average distance from point i to the closest cluster centroid
123 for (int j = 0; j < pointerCentroids.size(); ++j)
124 {
125 if (pointerCentroids[j] != *(points[i].centroid))
126 {
127 c += EuclideanMetric<PT, PD>::distanceTo(pointerCentroids[j], points[i]);
128 countC++;
129 }
130 }
131 if (countC > 0)
132 c /= countC; // Normalize c(i)
133
134 // Compute S(i): silhouette coefficient for the data point i
135 if (a > 0 || c > 0)
136 {
137 totalScore += (c - a) / std::max(a, c);
138 }
139 }
140
141 // Return the average silhouette score over all points
142 return totalScore / numPoints;
143}
