13/*
14 * Evaluate both global and local consistency error (GCE & LCE)
15 * see Martin et. al in ICCV 01'
16 */
17struct Entry_CE* EvaluateConsistencyError(Segmentation *s1, Segmentation *s2)
18{
19 int i, j;
20 double GCE = 0, LCE = 0, GCEa = 0, LCEa = 0;
21
22 // Number of segments in s1 and s2
23 int nSeg1 = s1->getSegments().size();
24 int nSeg2 = s2->getSegments().size();
25
26 // Number of faces in segments(should be the same for s1 and s2)
27 int nFaces = s1->getMesh()->numFaces();
28
29 // only compute under non-trial cases, otherwise, consistency errors are just 0
30 if (nSeg1 != 1 && nSeg2 != 1 && nSeg1 != nFaces && nSeg2 != nFaces)
31 {
32 // Allocate metrics to store the difference between sets of segments
33 // numFaces based
34 double *SetDifference12 = new double[nSeg1 * nSeg2]; // s1 sets over s2 sets
35 double *SetDifference21 = new double[nSeg2 * nSeg1]; // s2 sets over s1 sets
36 double *Intersection = new double[nSeg1 * nSeg2];
37 // area based
38 double *AreaDifference12 = new double[nSeg1 * nSeg2]; // s1 areas over s2 areas
39 double *AreaDifference21 = new double[nSeg2 * nSeg1]; // s2 areas over s1 areas
40 double *AreaIntersection = new double[nSeg1 * nSeg2];
41 for (i = 0; i < nSeg1 * nSeg2; i++)
42 {
43 // numFaces based
44 SetDifference12[i] = 0;
45 Intersection[i] = 0;
46 SetDifference21[i] = 0;
47 // area based
48 AreaDifference12[i] = 0;
49 AreaIntersection[i] = 0;
50 AreaDifference21[i] = 0;
51 }
52
53
54 // get intersections
55 for (FaceId face(0); face < s1->getMesh()->numFaces(); ++face)
56 {
57 int a = s1->getMesh()->getFaceCluster(face);
58 int b = s2->getMesh()->getFaceCluster(face);
59 Intersection[a * nSeg2 + b] += 1;
60 AreaIntersection[a * nSeg2 + b] += s1->getMesh()->getFace(face).getArea();
61 }
62
63 // get normalized set/area differences
64 // faces in segments[i] does not necessarily have the segment id i
65 double *nSegFaces1 = new double[nSeg1];
66 double *nSegFaces2 = new double[nSeg2];
67 double *areaSeg1 = new double[nSeg1];
68 double *areaSeg2 = new double[nSeg2];
69
70 for (i = 0; i < nSeg1; i++)
71 {
72 if (s1->getSegments()[i].getFaces().size() == 0)
73 {
74 nSegFaces1[i] = s1->getSegments()[i].getFaces().size();
75 areaSeg1[i] = s1->getSegments()[i].getArea();
76 } else {
77 FaceId f = s1->getSegments()[i].getFaces()[0];
78 int segId = s1->getMesh()->getFaceCluster(f);
79 nSegFaces1[segId] = s1->getSegments()[i].getFaces().size();
80 areaSeg1[segId] = s1->getSegments()[i].getArea();
81 }
82 }
83 for (i = 0; i < nSeg2; i++)
84 {
85 if (s2->getSegments()[i].getFaces().size() == 0) {
86 nSegFaces2[i] = s2->getSegments()[i].getFaces().size();
87 areaSeg2[i] = s2->getSegments()[i].getArea();
88 } else {
89 FaceId f = s2->getSegments()[i].getFaces()[0];
90 int segId = s2->getMesh()->getFaceCluster(f);
91 nSegFaces2[segId] = s2->getSegments()[i].getFaces().size();
92 areaSeg2[segId] = s2->getSegments()[i].getArea();
93 }
94 }
95
96 for (i = 0; i < nSeg1; i++)
97 {
98 for (j = 0; j < nSeg2; j++)
99 {
100 if (nSegFaces1[i] != 0)
101 SetDifference12[i * nSeg2 + j] = 1 - Intersection[i * nSeg2 + j] / nSegFaces1[i];
102 else
103 SetDifference12[i * nSeg2 + j] = 1;
104
105 if (nSegFaces2[j] != 0)
106 SetDifference21[j * nSeg1 + i] = 1 - Intersection[i * nSeg2 + j] / nSegFaces2[j];
107 else
108 SetDifference21[j * nSeg1 + i] = 1;
109
110 if (areaSeg1[i] != 0)
111 AreaDifference12[i * nSeg2 + j] = 1 - AreaIntersection[i * nSeg2 + j] / areaSeg1[i];
112 else
113 AreaDifference12[i * nSeg2 + j] = 1;
114
115 if (areaSeg2[j] != 0)
116 AreaDifference21[j * nSeg1 + i] = 1 - AreaIntersection[i * nSeg2 + j] / areaSeg2[j];
117 else
118 AreaDifference21[j * nSeg1 + i] = 1;
119 }
120 }
121
122 // get GCE/LCE/GCEa/LCEa
123 double sumE12 = 0, sumE21 = 0, sumMin = 0;
124 double sumEa12 = 0, sumEa21 = 0, sumMina = 0;
125 for (i = 0; i < nSeg1; i++)
126 {
127 for (j = 0; j < nSeg2; j++)
128 {
129 int frequency = int(Intersection[i * nSeg2 + j]);
130 sumE12 += SetDifference12[i * nSeg2 + j] * frequency;
131 sumE21 += SetDifference21[j * nSeg1 + i] * frequency;
132 sumMin += std::min(SetDifference12[i * nSeg2 + j], SetDifference21[j * nSeg1 + i]) * frequency;
133 sumEa12 += AreaDifference12[i * nSeg2 + j] * frequency;
134 sumEa21 += AreaDifference21[j * nSeg1 + i] * frequency;
135 sumMina += std::min(AreaDifference12[i * nSeg2 + j], AreaDifference21[j * nSeg1 + i]) * frequency;
136 }
137 }
138 // numFaces based
139 GCE = std::min(sumE12, sumE21) / nFaces;
140 LCE = sumMin / nFaces;
141 // area based
142 GCEa = std::min(sumEa12, sumEa21) / nFaces;
143 LCEa = sumMina / nFaces;
144
145 // Release memories
146 delete[] SetDifference12;
147 delete[] SetDifference21;
148 delete[] Intersection;
149 delete[] AreaDifference12;
150 delete[] AreaDifference21;
151 delete[] AreaIntersection;
152 delete[] nSegFaces1;
153 delete[] nSegFaces2;
154 delete[] areaSeg1;
155 delete[] areaSeg2;
156 }
157
158 // return the booking entry
159 Entry_CE *e = new Entry_CE;
160 e->GCE = GCE;
161 e->LCE = LCE;
162 e->GCEa = GCEa;
163 e->LCEa = LCEa;
164 return e;
165}
